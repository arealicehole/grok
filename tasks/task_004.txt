```python
class JobStatus(enum.Enum):
    PENDING = "pending"
    PROCESSING = "processing"
    COMPLETED = "completed"
    FAILED = "failed"

class AnalysisJob:
    def __init__(self, id=None, status=JobStatus.PENDING, transcript_id=None, profile_id=None, 
                 temporary_instructions=None, error_message=None, created_at=None, updated_at=None):
        # Validation logic
        if not transcript_id:
            raise ValueError("transcript_id is required")
        
        if not isinstance(status, JobStatus) and status not in [s.value for s in JobStatus]:
            raise ValueError(f"Invalid status: {status}")
            
        if not profile_id and not temporary_instructions:
            raise ValueError("Either profile_id or temporary_instructions must be provided")
            
        # Convert string status to enum if needed
        if isinstance(status, str):
            status = JobStatus(status)
            
        # Initialize fields
        self.id = id
        self.status = status
        self.transcript_id = transcript_id
        self.profile_id = profile_id
        self.temporary_instructions = temporary_instructions
        self.error_message = error_message
        self.created_at = created_at or datetime.datetime.now()
        self.updated_at = updated_at or datetime.datetime.now()
    
    @classmethod
    def from_dict(cls, data):
        """Create an AnalysisJob instance from a dictionary"""
        # Handle status conversion from string to enum
        status = data.get("status")
        if status and isinstance(status, str):
            try:
                status = JobStatus(status)
            except ValueError:
                raise ValueError(f"Invalid status value: {status}")
        
        return cls(
            id=data.get("id"),
            status=status or JobStatus.PENDING,
            transcript_id=data.get("transcript_id"),
            profile_id=data.get("profile_id"),
            temporary_instructions=data.get("temporary_instructions"),
            error_message=data.get("error_message"),
            created_at=data.get("created_at"),
```

## Request Payload Formatting

```python
def _prepare_request_payload(
    self, 
    transcript_text: str, 
    profile_instructions: str, 
    model: Optional[str] = None, 
    **kwargs
) -> Dict[str, Any]:
    """
    Format the input data into the structure expected by the Xai Grok API.
    
    Args:
        transcript_text: The transcript to analyze
        profile_instructions: Instructions for the analysis
        model: The model to use (defaults to self.default_model)
        **kwargs: Additional parameters to include in the request
        
    Returns:
        Dict containing the formatted request payload
    """
    if not transcript_text:
        raise ValueError("Transcript text cannot be empty")
    
    # Create message structure
    messages = [
        {"role": "system", "content": profile_instructions},
        {"role": "user", "content": transcript_text}
    ]
    
    # Build the payload
    payload = {
        "model": model or self.default_model,
        "messages": messages,
        # Default parameters
        "temperature": 0.7,
        "max_tokens": 1024,
    }
    
    # Override defaults with any provided kwargs
    payload.update(kwargs)
    
    self._log_request("POST", self.api_endpoint, payload)
    return payload
```

## API Communication with Retry Logic

```python
def send_analysis_request(
    self, 
    transcript_text: str, 
    profile_instructions: str, 
    model: Optional[str] = None, 
    retries: int = settings.MAX_RETRIES, 
    **kwargs
) -> Dict[str, Any]:
    """
    Send a request to the Xai Grok API with retry logic and error handling.
    
    Args:
        transcript_text: The transcript to analyze
        profile_instructions: Instructions for the analysis
        model: The model to use (defaults to self.default_model)
        retries: Number of retry attempts for transient errors
        **kwargs: Additional parameters to include in the request
        
    Returns:
        Dict containing the API response
        
    Raises:
        AuthenticationError: If API key is invalid
        RateLimitError: If rate limits are exceeded
        APIError: For other API errors
        ConnectionError: For network issues
    """
    payload = self._prepare_request_payload(
        transcript_text, profile_instructions, model, **kwargs
    )
    
    # Retry logic with exponential backoff
    for attempt in range(retries + 1):
        try:
            response = self.session.post(
                self.api_endpoint,
                json=payload,
                timeout=self.timeout
            )
            
            # Handle successful response
            if response.status_code == 200:
                result = response.json()
                self._log_response(response)
                return result
                
            # Handle rate limiting with retry
            if response.status_code == 429:
                if attempt < retries:
                    wait_time = 2 ** attempt + 1  # Exponential backoff
                    self.logger.warning(
                        f"Rate limit exceeded. Retrying in {wait_time}s. "
                        f"Attempt {attempt + 1}/{retries}"
                    )
                    time.sleep(wait_time)
                    continue
                else:
                    return self._handle_api_error(response)
            
            # Handle other errors
            return self._handle_api_error(response)
            
        except requests.exceptions.Timeout:
            self.logger.error(f"Request timed out. Attempt {attempt + 1}/{retries}")
            if attempt >= retries:
                raise ConnectionError("Request timed out after multiple attempts")
        
        except requests.exceptions.ConnectionError as e:
            self.logger.error(f"Connection error: {str(e)}. Attempt {attempt + 1}/{retries}")
            if attempt >= retries:
                raise ConnectionError(f"Failed to connect to API: {str(e)}")
        
        # Exponential backoff before retry
        if attempt < retries:
            time.sleep(2 ** attempt)
```

## Error Handling and Logging

```python
def _handle_api_error(self, response) -> Dict[str, Any]:
    """Handle API error responses and raise appropriate exceptions"""
    error_msg = f"API Error {response.status_code}: "
    
    try:
        error_data = response.json()
        error_msg += error_data.get('error', {}).get('message', 'Unknown error')
    except ValueError:
        error_msg += response.text or "No error details available"
    
    self.logger.error(error_msg)
    
    if response.status_code == 401:
        raise AuthenticationError("Invalid API key or authentication failure")
    elif response.status_code == 403:
        raise AuthorizationError("Not authorized to access this resource")
    elif response.status_code == 429:
        raise RateLimitError("Rate limit exceeded")
    elif 400 <= response.status_code < 500:
        raise InvalidRequestError(error_msg)
    else:
        raise APIError(error_msg)

def _log_request(self, method, url, payload):
    """Log API request details (redacting sensitive information)"""
    # Create a copy to avoid modifying the original payload
    safe_payload = payload.copy() if payload else {}
    
    # Redact sensitive information if present
    if 'api_key' in safe_payload:
        safe_payload['api_key'] = '***REDACTED***'
    
    self.logger.debug(
        f"API Request: {method} {url}\n"
        f"Payload: {safe_payload}"
    )

def _log_response(self, response):
    """Log API response details"""
    self.logger.debug(
        f"API Response: {response.status_code}\n"
        f"Headers: {dict(response.headers)}\n"
        f"Content: {response.text[:500]}{'...' if len(response.text) > 500 else ''}"
    )
```

```python
def _chunk_transcript(text, max_chunk_size=8000):
    """Split large transcripts into manageable chunks"""
    # Simple sentence-based chunking
    sentences = re.split(r'(?<=[.!?])\s+', text)
    chunks = []
    current_chunk = ""
    
    for sentence in sentences:
        if len(current_chunk) + len(sentence) <= max_chunk_size:
            current_chunk += sentence + " "
        else:
            chunks.append(current_chunk.strip())
            current_chunk = sentence + " "
    
    if current_chunk:
        chunks.append(current_chunk.strip())
    
    return chunks
```

```python
def parse_analysis_response(raw_response, analysis_type):
    """
    Parse and validate API response based on analysis type.
    
    Args:
        raw_response (str or dict): Raw API response
        analysis_type (str): Type of analysis performed
        
    Returns:
        dict: Validated and structured analysis data
        
    Raises:
        ResponseParsingError: If response cannot be parsed
        ResponseValidationError: If response fails schema validation
    """
    try:
        # Convert string to dict if needed
        if isinstance(raw_response, str):
            data = json.loads(raw_response)
        else:
            data = raw_response
            
        # Extract content from Grok API response structure
        if "choices" in data and len(data["choices"]) > 0:
            if "message" in data["choices"][0] and "content" in data["choices"][0]["message"]:
                content = data["choices"][0]["message"]["content"]
                
                # Handle different response formats (JSON string vs structured text)
                try:
                    parsed_content = json.loads(content)
                except json.JSONDecodeError:
                    # If not JSON, apply text parsing based on analysis type
                    parsed_content = _parse_text_response(content, analysis_type)
                
                # Validate against schema
                schema = ANALYSIS_SCHEMAS.get(analysis_type)
                if schema:
                    jsonschema.validate(parsed_content, schema)
                
                # Apply transformations specific to analysis type
                return _transform_response(parsed_content, analysis_type)
            
        raise ResponseParsingError("Unable to extract content from API response")
    except json.JSONDecodeError as e:
        raise ResponseParsingError(f"Invalid JSON format: {str(e)}")
    except jsonschema.exceptions.ValidationError as e:
        raise ResponseValidationError(f"Response failed schema validation: {str(e)}")
```

### Data Sanitization

```python
def _sanitize_response_data(data):
    """
    Sanitize response data to prevent security issues.
    
    Args:
        data (dict): Parsed response data
        
    Returns:
        dict: Sanitized data
    """
    sanitized = {}
    
    # Recursively process dictionary
    for key, value in data.items():
        # Convert keys to snake_case and sanitize
        safe_key = re.sub(r'[^\w]', '_', key.lower())
        
        if isinstance(value, dict):
            sanitized[safe_key] = _sanitize_response_data(value)
        elif isinstance(value, list):
            sanitized[safe_key] = [
                _sanitize_response_data(item) if isinstance(item, dict) 
                else html.escape(str(item)) if isinstance(item, str)
                else item
                for item in value
            ]
        elif isinstance(value, str):
            # Escape HTML entities to prevent XSS
            sanitized[safe_key] = html.escape(value)
        else:
            # Numbers, booleans, etc. pass through
            sanitized[safe_key] = value
            
    return sanitized
```

```python
def recover_partial_response(raw_response, analysis_type):
    """
    Attempt to extract useful data from malformed responses.
    
    Args:
        raw_response (str or dict): Raw API response
        analysis_type (str): Type of analysis performed
        
    Returns:
        dict: Partially recovered data or empty dict if recovery fails
    """
    try:
        # Attempt to extract any JSON-like structures from text
        if isinstance(raw_response, str):
            # Find JSON-like patterns using regex
            json_pattern = r'\{(?:[^{}]|(?R))*\}'
            matches = re.findall(json_pattern, raw_response)
            
            for potential_json in matches:
                try:
                    data = json.loads(potential_json)
                    # Check if it has at least some expected keys
                    schema = ANALYSIS_SCHEMAS.get(analysis_type, {})
                    required_keys = schema.get("required", [])
                    
                    if any(key in data for key in required_keys):
                        return {
                            "recovered_data": data,
                            "is_complete": False,
                            "missing_fields": [k for k in required_keys if k not in data]
                        }
                except:
                    continue
                    
        return {}
    except Exception:
        return {}
```

```python
def handle_response_errors(raw_response, analysis_type, job_id=None):
    """
    Comprehensive error handling for API responses with recovery attempts.
    
    Args:
        raw_response (str or dict): Raw API response
        analysis_type (str): Type of analysis performed
        job_id (int, optional): Associated job ID for logging
        
    Returns:
        tuple: (success_flag, result_dict or error_message)
    """
    try:
        result = parse_analysis_response(raw_response, analysis_type)
        return True, result
    except ResponseParsingError as e:
        logger.error(f"Parsing error for job {job_id}: {str(e)}")
        
        # Attempt recovery
        recovered_data = recover_partial_response(raw_response, analysis_type)
        if recovered_data:
            logger.warning(f"Partially recovered data for job {job_id}")
            return False, {
                "partial_data": recovered_data,
                "error": str(e),
                "status": "partial_recovery"
            }
        
        return False, {"error": str(e), "status": "parsing_failed"}
    except ResponseValidationError as e:
        logger.error(f"Validation error for job {job_id}: {str(e)}")
        
        # Extract what data we can, even if it doesn't match schema
        if isinstance(raw_response, str):
            try:
                data = json.loads(raw_response)
                return False, {
                    "unvalidated_data": data,
                    "error": str(e),
                    "status": "validation_failed"
                }
            except:
                pass
                
        return False, {"error": str(e), "status": "validation_failed"}
```

### Malformed Response Detection
```python
def detect_malformed_response(response_data):
    """
    Detect common patterns of malformed responses.
    
    Args:
        response_data (dict): Parsed response data
        
    Returns:
        tuple: (is_malformed, issue_description)
    """
    # Check for empty or null values in required fields
    if not response_data:
        return True, "Empty response data"
        
    # Check for truncated JSON (common with token limits)
    if isinstance(response_data, str) and response_data.count('{') != response_data.count('}'):
        return True, "Truncated JSON structure"
        
    # Check for common error patterns in LLM responses
    error_patterns = [
        "I'm sorry", "I apologize", "I cannot", "As an AI",
        "error", "unable to", "failed to"
    ]
    
    if isinstance(response_data, str):
        lower_response = response_data.lower()
        for pattern in error_patterns:
            if pattern.lower() in lower_response:
                return True, f"Response contains error indicator: '{pattern}'"
                
    return False, None
```

```python
def _parse_text_response(text_content, analysis_type):
    """
    Parse non-JSON text responses based on analysis type.
    
    Args:
        text_content (str): Text content from API response
        analysis_type (str): Type of analysis performed
        
    Returns:
        dict: Structured data extracted from text
    """
    result = {}
    
    if analysis_type == "sentiment":
        # Extract sentiment patterns
        sentiment_match = re.search(r'sentiment[:\s]+([a-z]+)', text_content, re.IGNORECASE)
        confidence_match = re.search(r'confidence[:\s]+([\d.]+)', text_content, re.IGNORECASE)
        
        if sentiment_match:
            result["sentiment"] = sentiment_match.group(1).lower()
        
        if confidence_match:
            try:
                result["confidence"] = float(confidence_match.group(1))
            except ValueError:
                result["confidence"] = 0.0
        
        # Extract explanation (everything after "explanation:" if present)
        explanation_match = re.search(r'explanation[:\s]+(.*?)(?:\n\n|\Z)', 
                                     text_content, re.IGNORECASE | re.DOTALL)
        if explanation_match:
            result["explanation"] = explanation_match.group(1).strip()
        else:
            # Use the whole text as explanation if no specific section found
            result["explanation"] = text_content
    
    elif analysis_type == "topic":
        # Extract topics list
        topics = []
        topic_section = re.search(r'topics[:\s]+(.*?)(?:\n\n|\Z)', 
                                 text_content, re.IGNORECASE | re.DOTALL)
        
        if topic_section:
            # Look for list items
            topic_items = re.findall(r'[-*•]\s*(.*?)(?:\n|$)', topic_section.group(1))
            if topic_items:
                topics = [item.strip() for item in topic_items]
            else:
                # Split by commas or newlines if no list markers
                topics = [t.strip() for t in re.split(r',|\n', topic_section.group(1)) if t.strip()]
        
        result["topics"] = topics
    
    # Add fallback values for required fields if missing
    if analysis_type == "sentiment":
        if "sentiment" not in result:
            result["sentiment"] = "neutral"
        if "confidence" not in result:
            result["confidence"] = 0.5
        if "explanation" not in result:
            result["explanation"] = "No explanation provided"
    
    return result

def _transform_response(parsed_content, analysis_type):
    """
    Transform parsed content into standardized format based on analysis type.
    
    Args:
        parsed_content (dict): Parsed content from API response
        analysis_type (str): Type of analysis performed
        
    Returns:
        dict: Transformed data in standardized format
    """
    transformed = {}
    
    if analysis_type == "sentiment":
        # Standardize sentiment values
        if "sentiment" in parsed_content:
            sentiment = parsed_content["sentiment"].lower()
            # Map various sentiment expressions to standard values
            sentiment_mapping = {
                "positive": ["positive", "good", "favorable", "happy", "satisfied"],
                "negative": ["negative", "bad", "unfavorable", "unhappy", "dissatisfied"],
                "neutral": ["neutral", "mixed", "balanced", "unclear"]
            }
            
            for standard, variations in sentiment_mapping.items():
                if sentiment in variations:
                    transformed["sentiment"] = standard
                    break
            else:
                # Default if no match found
                transformed["sentiment"] = "neutral"
        
        # Normalize confidence to 0-1 range
        if "confidence" in parsed_content:
            confidence = parsed_content["confidence"]
            if isinstance(confidence, (int, float)):
                # Ensure value is between 0 and 1
                transformed["confidence"] = max(0.0, min(1.0, float(confidence)))
            elif isinstance(confidence, str):
                # Try to convert string to float
                try:
                    confidence_value = float(confidence.strip('%')) / 100 if '%' in confidence else float(confidence)
                    transformed["confidence"] = max(0.0, min(1.0, confidence_value))
                except ValueError:
                    transformed["confidence"] = 0.5
            else:
                transformed["confidence"] = 0.5
        
        # Copy other fields
        for field in ["explanation", "key_phrases"]:
            if field in parsed_content:
                transformed[field] = parsed_content[field]
    
    # Copy any remaining fields
    for key, value in parsed_content.items():
        if key not in transformed:
            transformed[key] = value
    
    return transformed
```

```python
# src/core/decorators.py
import functools
import logging
from typing import Type, List, Optional, Callable, Any

from .exceptions import AnalysisError, RetryableError

logger = logging.getLogger(__name__)

def error_handler(
    retryable_exceptions: List[Type[Exception]] = None,
    max_retries: int = 3,
    update_job: bool = True,
    log_level: int = logging.ERROR
):
    """
    Decorator for handling errors in analysis functions.
    
    Args:
        retryable_exceptions: List of exception types that should trigger retries
        max_retries: Maximum number of retry attempts
        update_job: Whether to update the AnalysisJob status
        log_level: Logging level for errors
    """
    retryable_exceptions = retryable_exceptions or [RetryableError]
    
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            job = kwargs.get('job')
            if not job and update_job:
                # Try to find job in args
                for arg in args:
                    if hasattr(arg, 'status') and hasattr(arg, 'error_message'):
                        job = arg
                        break
            
            retry_count = 0
            last_error = None
            
            while retry_count <= max_retries:
                try:
                    return func(*args, **kwargs)
                except tuple(retryable_exceptions) as e:
                    retry_count += 1
                    last_error = e
                    
                    error_msg = f"{func.__name__} failed with {type(e).__name__}: {str(e)}"
                    logger.log(log_level, error_msg)
                    
                    if retry_count <= max_retries:
                        logger.info(f"Retrying {func.__name__} (attempt {retry_count}/{max_retries})")
                    else:
                        logger.error(f"Max retries ({max_retries}) exceeded for {func.__name__}")
                        if job and update_job:
                            job.status = "failed"
                            job.error_message = f"Max retries exceeded: {str(e)}"
                        raise
                except Exception as e:
                    error_msg = f"{func.__name__} failed with {type(e).__name__}: {str(e)}"
                    logger.log(log_level, error_msg)
                    
                    if job and update_job:
                        job.status = "failed"
                        job.error_message = str(e)
                    
                    raise
            
            return None  # Should never reach here
        return wrapper
    return decorator
```

## Exponential Backoff Implementation (Point 4)

```python
# src/core/utils.py
import random
import time
from typing import Optional, Callable, Any

def exponential_backoff_retry(
    func: Callable,
    retryable_exceptions: tuple,
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    jitter: bool = True
) -> Any:
    """
    Execute a function with exponential backoff retry logic.
    
    Args:
        func: Function to execute
        retryable_exceptions: Tuple of exceptions that should trigger a retry
        max_retries: Maximum number of retry attempts
        base_delay: Initial delay in seconds
        max_delay: Maximum delay in seconds
        jitter: Whether to add randomness to the delay
        
    Returns:
        The return value of the function
        
    Raises:
        The last exception if max_retries is exceeded
    """
    last_exception = None
    
    for attempt in range(max_retries + 1):
        try:
            return func()
        except retryable_exceptions as e:
            last_exception = e
            
            if attempt == max_retries:
                raise
            
            # Calculate delay with exponential backoff
            delay = min(base_delay * (2 ** attempt), max_delay)
            
            # Add jitter if enabled (helps prevent thundering herd)
            if jitter:
                delay = delay * (0.5 + random.random())
                
            time.sleep(delay)
    
    # Should never reach here
    raise last_exception if last_exception else RuntimeError("Unknown error in retry logic")
```

```python
def general_error_handler(update_job_on_error=True, log_level=logging.ERROR):
    """
    Decorator for standardized error handling across the application.
    
    Args:
        update_job_on_error: Whether to update the AnalysisJob status on error
        log_level: Logging level for errors
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            job = kwargs.get('job')
            if not job and update_job_on_error:
                # Try to find job in args
                for arg in args:
                    if hasattr(arg, 'status') and hasattr(arg, 'error_message'):
                        job = arg
                        break
            
            try:
                return func(*args, **kwargs)
            except Exception as e:
                # Log the error
                error_msg = f"{func.__name__} failed: {type(e).__name__}: {str(e)}"
                logger.log(log_level, error_msg, exc_info=True)
                
                # Update job if available
                if job and update_job_on_error:
                    job.status = "FAILED"
                    job.error_message = str(e)
                    # Store raw exception for debugging
                    setattr(job, '_raw_exception', e)
                
                # Re-raise as AnalysisProcessingError if it's not already
                if not isinstance(e, AnalysisProcessingError):
                    raise AnalysisProcessingError(error_msg) from e
                raise
                
        return wrapper
    return decorator
```
