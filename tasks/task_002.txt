# Example implementation of the AnalysisProfile class
import uuid
from datetime import datetime
import json
from typing import Dict, Any, Optional

class AnalysisProfile:
    def __init__(
        self,
        name: str,
        instructions: str = "",
        schema_definition: Dict[str, Any] = None,
        id: Optional[uuid.UUID] = None,
        created_at: Optional[datetime] = None,
        updated_at: Optional[datetime] = None
    ):
        self.id = id or uuid.uuid4()
        self.name = name
        self.instructions = instructions
        self.schema_definition = schema_definition or {}
        self.created_at = created_at or datetime.utcnow()
        self.updated_at = updated_at or self.created_at
        
        # Validate on initialization
        if not self.is_valid():
            raise ValueError("Invalid AnalysisProfile parameters")
    
    def is_valid(self) -> bool:
        """Validate the profile has required fields and proper structure"""
        if not self.name or not isinstance(self.name, str):
            return False
            
        # Validate schema structure - check for required components
        try:
            if self.schema_definition:
                # Ensure schema has input and output sections
                if not all(k in self.schema_definition for k in ['input', 'output']):
                    return False
        except Exception:
            return False
            
        return True
        
    def update(self, **kwargs) -> None:
        """Update profile attributes and set updated_at timestamp"""
        for key, value in kwargs.items():
            if hasattr(self, key):
                setattr(self, key, value)
        self.updated_at = datetime.utcnow()
    
    def to_dict(self) -> Dict[str, Any]:
        """Convert profile to dictionary for serialization"""
        return {
            'id': str(self.id),
            'name': self.name,
            'instructions': self.instructions,
            'schema_definition': self.schema_definition,
            'created_at': self.created_at.isoformat(),
            'updated_at': self.updated_at.isoformat()
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'AnalysisProfile':
        """Create profile from dictionary representation"""
        return cls(
            id=uuid.UUID(data['id']) if isinstance(data.get('id'), str) else data.get('id'),
            name=data['name'],
            instructions=data.get('instructions', ''),
            schema_definition=data.get('schema_definition', {}),
            created_at=datetime.fromisoformat(data['created_at']) if isinstance(data.get('created_at'), str) else data.get('created_at'),
            updated_at=datetime.fromisoformat(data['updated_at']) if isinstance(data.get('updated_at'), str) else data.get('updated_at')
        )

    """

    Args:

        new_name: Optional new name (defaults to "Copy of [original name]")
        modifications: Dictionary of attributes to modify in the new profile
        
    Returns:
        The newly created profile
    """
    # Implementation example with deep copying to avoid reference issues
    import copy
    source = load_profile(source_id)
    if not source:
        raise ValueError(f"Source profile with ID {source_id} not found")
        
    # Deep copy to avoid reference issues with nested objects
    new_profile = copy.deepcopy(source)

def validate_schema_compatibility(schema1: dict, schema2: dict) -> Tuple[bool, List[str]]:
    """
    Checks if two schemas are compatible for data exchange or merging.
    
    Returns:
        Tuple containing (is_compatible, list_of_incompatibilities)
    """
    incompatibilities = []
    
    # Check schema versions
    if schema1.get("version") != schema2.get("version"):
        incompatibilities.append(f"Version mismatch: {schema1.get('version')} vs {schema2.get('version')}")
    
    # Check required fields
    schema1_required = {f["name"] for f in schema1.get("fields", []) if f.get("required")}
    schema2_required = {f["name"] for f in schema2.get("fields", []) if f.get("required")}
    
    # Check if schema2 has all required fields from schema1
    missing_required = schema1_required - {f["name"] for f in schema2.get("fields", [])}
    if missing_required:
        incompatibilities.append(f"Missing required fields: {', '.join(missing_required)}")
    
    # Check field type compatibility
    schema1_fields = {f["name"]: f["type"] for f in schema1.get("fields", [])}
    schema2_fields = {f["name"]: f["type"] for f in schema2.get("fields", [])}

class StorageCircuitBreaker:
    def __init__(self, failure_threshold=5, reset_timeout=60):
        self.failure_count = 0
        self.failure_threshold = failure_threshold
        self.reset_timeout = reset_timeout
        self.last_failure_time = 0
        self.state = "CLOSED"  # CLOSED, OPEN, HALF-OPEN
        self.lock = threading.RLock()
    
    def execute(self, func, *args, **kwargs):
        with self.lock:
            current_time = time.time()
            if self.state == "OPEN":
                if current_time - self.last_failure_time > self.reset_timeout:
                    self.state = "HALF-OPEN"
                else:
                    raise ProfileStorageConnectionError("Circuit breaker open - storage system unavailable")
            
            try:
                result = func(*args, **kwargs)
                if self.state == "HALF-OPEN":
                    self.state = "CLOSED"
                    self.failure_count = 0
                return result
            except (ProfileStorageConnectionError, ProfileCorruptionError) as e:
                self.failure_count += 1
                self.last_failure_time = current_time
                if self.failure_count >= self.failure_threshold:
                    self.state = "OPEN"
                raise

            # Trim old timestamps (keep last 100)

                self.error_timestamps[error_type] = self.error_timestamps[error_type][-100:]

        with self.lock:
            now = time.time()
            recent_errors = [t for t in self.error_timestamps[error_type] 
                            if now - t <= time_window]

                "counts": dict(self.error_counts),

import time

def handle_db_errors(func):
    """Decorator to handle database errors with retry logic for locks"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        max_retries = 3
        retry_count = 0
        
        while True:
            try:
                return func(*args, **kwargs)
            except sqlite3.OperationalError as e:
                error_msg = str(e).lower()
                if "database is locked" in error_msg and retry_count < max_retries:
                    # Exponential backoff for database locks
                    retry_count += 1
                    wait_time = 0.2 * (2 ** (retry_count - 1))
                    logging.warning(
                        f"Database locked, retrying in {wait_time:.2f}s (attempt {retry_count}/{max_retries})"
                    )
                    time.sleep(wait_time)
                    continue
                else:
                    logging.error(f"Database error in {func.__name__}: {e}", exc_info=True)
                    raise map_sqlite_exception(e)
            except sqlite3.Error as e:
                logging.error(f"Database error in {func.__name__}: {e}", exc_info=True)
                raise map_sqlite_exception(e)
            except json.JSONDecodeError as e:
                logging.error(f"JSON decode error in {func.__name__}: {e}", exc_info=True)
                raise ProfileCorruptionError(f"Invalid JSON data: {e}")
            except Exception as e:
                logging.error(f"Unexpected error in {func.__name__}: {e}", exc_info=True)
                raise
    return wrapper

class DatabaseLockMonitor:
    def __init__(self):
        self.lock_events = []

    def record_lock_event(self, operation_name, retry_count):
        with self.lock:
            self.lock_events.append({
                "timestamp": time.time(),
                "operation": operation_name,
                "retries": retry_count
            })
            # Keep only the last 100 events
            if len(self.lock_events) > 100:
                self.lock_events = self.lock_events[-100:]
    
    def get_lock_frequency(self, time_window=300):
        """Get number of lock events in the last time_window seconds"""
        with self.lock:
            now = time.time()
            recent_locks = [e for e in self.lock_events 
                           if now - e["timestamp"] <= time_window]
            return len(recent_locks)
            
    def get_problematic_operations(self):
        """Identify operations with the most lock conflicts"""
        with self.lock:
            operation_counts = {}
            for event in self.lock_events:
                op = event["operation"]
                operation_counts[op] = operation_counts.get(op, 0) + 1
            return sorted(operation_counts.items(), key=lambda x: x[1], reverse=True)

   def compute_schema_checksum(schema_dict):

   from jsonschema import Draft7Validator, exceptions as schema_exceptions

   def validate_schema_against_standard(schema_definition):
       """
