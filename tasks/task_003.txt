```python
def _validate_form(self):
    # Get form values
    name = self.name_entry.get().strip()
    instructions = self.instructions_text.get("1.0", "end-1c").strip()
    
    # Validation rules
    if not name:
        return False, "Profile name is required"
    
    if len(name) > 50:
        return False, "Profile name must be 50 characters or less"
    
    # Check for duplicate names (excluding current profile)
    existing = self.repository.get_profile_by_name(name)
    if existing and (not self.profile_id or existing.id != self.profile_id):
        return False, "A profile with this name already exists"
        
    return True, None
```

```python
class ProfileEditorWindow(ctk.CTkToplevel):
    def __init__(self, master, repository, profile_id=None, on_save_callback=None):
        # Initialize state variables
        self.original_data = {}  # Store original profile data for comparison
        self.current_tags = set()  # Track selected tags
        
    def _track_changes(self):
        """Compare current form state with original data to determine if changes exist"""
        current_name = self.name_entry.get().strip()
        current_instructions = self.instructions_text.get("1.0", "end-1c").strip()
        
        # Check if any field has changed from original
        return (current_name != self.original_data.get('name', '') or
                current_instructions != self.original_data.get('instructions', '') or
                self.current_tags != set(self.original_data.get('tags', [])))
```

```python
def _save_profile(self):
    try:
        valid, error_msg = self._validate_form()
        if not valid:
            CTkMessagebox(title="Validation Error", message=error_msg, icon="cancel")
            return False
            
        # Create/update profile object
        profile_data = {
            'name': self.name_entry.get().strip(),
            'instructions': self.instructions_text.get("1.0", "end-1c").strip(),
            'tags': list(self.current_tags)
        }
        
        if self.profile_id:
            profile_data['id'] = self.profile_id
            
        success = self.repository.save_profile(profile_data)
        if not success:
            CTkMessagebox(title="Save Error", message="Failed to save profile", icon="cancel")
            return False
            
        if self.on_save_callback:
            self.on_save_callback()
            
        self.destroy()
        return True
        
    except Exception as e:
        CTkMessagebox(title="Error", message=f"An error occurred: {str(e)}", icon="cancel")
        return False
```

```python
class AnalysisProfile:
    def __init__(self, id=None, name="", instructions="", tags=None):
```

```python
def _adjust_for_screen_size(self):
    """Adjust editor size based on screen dimensions"""
    # Get screen width and height
    screen_width = self.winfo_screenwidth()
    screen_height = self.winfo_screenheight()
    
    # Calculate dimensions (responsive to screen size)
    width = min(600, int(screen_width * 0.6))
    height = min(500, int(screen_height * 0.7))
    
    # Center window
    x = (screen_width - width) // 2
    y = (screen_height - height) // 2
    
    # Set geometry
    self.geometry(f"{width}x{height}+{x}+{y}")
```

```python
def _refresh_profile_list(self):
    """Refresh the profile list and maintain selection if possible"""
    # Store currently selected profile name if any
    selected_idx = self.profile_listbox.curselection()
    selected_name = None
    if selected_idx:
        selected_name = self.profile_listbox.get(selected_idx[0])
    
    # Clear and reload the list
    self.profile_listbox.delete(0, tk.END)
    profiles = self.profile_repository.get_all_profiles()
    
    # Sort profiles alphabetically by name
    profiles.sort(key=lambda p: p.name.lower())
    
    # Populate list and try to restore selection
    new_selected_idx = 0
    for i, profile in enumerate(profiles):
        self.profile_listbox.insert(tk.END, profile.name)
        if profile.name == selected_name:
            new_selected_idx = i
    
    # Restore selection if possible
    if profiles and selected_name:
        self.profile_listbox.selection_set(new_selected_idx)
        self.profile_listbox.see(new_selected_idx)
```
